def wordrank_extractor(textline, beta = 0.85, max_iter=10):
    wordrank_extractor_mini = KRWordRank(
        min_count = 5, # 단어의 최소 출현 빈도수 (그래프 생성 시)
        max_length = 10, # 단어의 최대 길이
        verbose = True
    )
    try:
        a,b,c = wordrank_extractor_mini.extract(textline, beta, max_iter)
    except AttributeError as e: 
        pass
    except ZeroDivisionError as f:
        return ''
        
    return a,b,c

def get_articles(data,beta, max_iter):
    for i in range(len(data.article)):
        #기사 하나씩 Sentence를 빼줍니다.
        k = Kkma()
        data = data.reset_index(drop=True)
        texts = k.sentences(data.article[i])
        texts2= [normalize(text, english=True, number=True) for text in texts]
        print(i)
        #기사 하나의 중요한 단어들이 나옵니다. 
        try:
            keywords, rank, graph = wordrank_extractor(texts2, beta, max_iter)
        except ValueError:
            pass
        
        stopwords = {'있다', '있는','대한','것이','인포','한다','제품','서울','우리','지난','옥상','미래'}
        passwords = {word:score for word, score in sorted(keywords.items(), key=lambda x:-x[1])[:300] if not (word in stopwords)}
        
        
        #해당 단어들을 total에 저장합니다. 
        for word in sorted(passwords, key=lambda x:x[1], reverse=True)[:30]:
            if word in total: #만약 원래 있는 애들이었다면, 기존 점수에 더해줍니다. 
                total[word] = total[word]+passwords[word]
            else: #원래 있지 않았다면, 새로 만들어줍니다. 
                total[word] = passwords[word]
    return total
